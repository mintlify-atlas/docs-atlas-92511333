The Parts Catalog uses Vue Router 3.5.1 for client-side routing with HTML5 history mode.

## Router Configuration

The router is configured in `src/router.js` with the following setup:

```javascript
import Vue from "vue";
import Router from "vue-router";

Vue.use(Router);

export default new Router({
  mode: "history",
  routes: [
    // Route definitions
  ]
});
```

**Key Configuration:**
- **Mode**: `history` - Uses HTML5 History API for clean URLs without hash (#)
- **Lazy Loading**: All components are loaded dynamically using `import()`

## Navigation Methods

### Declarative Navigation

Using `<router-link>` in templates:

```vue
<template>
  <!-- Navigate to parts list -->
  <router-link to="/parts">View Parts</router-link>
  
  <!-- Navigate to add part -->
  <router-link to="/add">Add Part</router-link>
  
  <!-- Navigate with route name -->
  <router-link :to="{ name: 'vehicles' }">Vehicles</router-link>
  
  <!-- Navigate with dynamic parameter -->
  <a :href="'/parts/' + currentPart.id">Edit Part</a>
</template>
```

### Programmatic Navigation

Using `this.$router.push()` in component methods:

```javascript
// Navigate to parts list after update
updatePart() {
  PartDataService.update(this.currentPart.id, this.currentPart)
    .then(() => {
      this.message = "Peça atualizada com sucesso!";
      this.$router.push({ name: "parts" });
    })
    .catch(e => {
      console.error(e);
    });
}

// Navigate after delete
deletePart() {
  PartDataService.delete(this.currentPart.id)
    .then(() => {
      this.$router.push({ name: "parts" });
    })
    .catch(e => {
      console.error(e);
    });
}
```

## Data Services Integration

Routes work seamlessly with data services to fetch and manipulate data.

### PartDataService

Location: `src/services/PartDataService.js`

```javascript
import http from "../http-common";

class PartDataService {
  getAll() {
    return http.get("/parts");
  }

  get(id) {
    return http.get(`/parts/${id}`);
  }

  create(data) {
    return http.post("/parts", data);
  }

  update(id, data) {
    return http.put(`/parts/${id}`, data);
  }

  delete(id) {
    return http.delete(`/parts/${id}`);
  }

  deleteAll() {
    return http.delete(`/parts`);
  }

  findByName(name) {
    return http.get(`/parts?name=${name}`);
  }

  findByVehicle(vehicle) {
    return http.get(`/parts?vehicle=${vehicle}`);
  }
}

export default new PartDataService();
```

### VehicleDataService

Location: `src/services/VehicleDataService.js`

```javascript
import http from "../http-common";

class VehicleDataService {
  getAll() {
    return http.get("/vehicles");
  }

  get(id) {
    return http.get(`/vehicles/${id}`);
  }

  create(data) {
    return http.post("/vehicles", data);
  }

  update(id, data) {
    return http.put(`/vehicles/${id}`, data);
  }

  delete(id) {
    return http.delete(`/vehicles/${id}`);
  }

  deleteAll() {
    return http.delete(`/vehicles`);
  }

  findByName(name) {
    return http.get(`/vehicles?name=${name}`);
  }

  findByPart(partId) {
    return http.get(`/parts/${partId}/vehicles`);
  }

  addToPart(partId, vehicleId) {
    return http.post(`/parts/${partId}/vehicles/${vehicleId}`);
  }

  removeFromPart(partId, vehicleId) {
    return http.delete(`/parts/${partId}/vehicles/${vehicleId}`);
  }
}

export default new VehicleDataService();
```

## HTTP Client Configuration

All data services use a centralized Axios configuration.

Location: `src/http-common.js`

```javascript
import axios from "axios";

export default axios.create({
  baseURL: "http://localhost:8080/api/v1",
  headers: {
    "Content-type": "application/json"
  }
});
```

**Configuration Details:**
- **Base URL**: `http://localhost:8080/api/v1`
- **Headers**: JSON content type
- **Export**: Default Axios instance used by all services

<Warning>
  Update the `baseURL` in `http-common.js` if your backend runs on a different host or port.
</Warning>

## API Endpoint Mapping

### Parts Endpoints

| Route | HTTP Method | API Endpoint | Purpose |
|-------|-------------|--------------|----------|
| `/` | GET | `/parts` | List all parts |
| `/parts/:id` | GET | `/parts/:id` | Get part details |
| `/add` | POST | `/parts` | Create new part |
| `/parts/:id` | PUT | `/parts/:id` | Update part |
| `/parts/:id` | DELETE | `/parts/:id` | Delete part |

### Vehicles Endpoints

| Route | HTTP Method | API Endpoint | Purpose |
|-------|-------------|--------------|----------|
| `/vehicles` | GET | `/vehicles` | List all vehicles |
| `/vehicles/add` | POST | `/vehicles` | Create new vehicle |
| - | GET | `/parts/:partId/vehicles` | Get vehicles for a part |
| - | POST | `/parts/:partId/vehicles/:vehicleId` | Link vehicle to part |
| - | DELETE | `/parts/:partId/vehicles/:vehicleId` | Unlink vehicle from part |

## Usage Examples

### Example 1: Fetching Parts on Route Load

```javascript
export default {
  name: "parts-list",
  data() {
    return {
      parts: []
    };
  },
  methods: {
    retrieveParts() {
      PartDataService.getAll()
        .then(response => {
          this.parts = response.data;
        })
        .catch(e => {
          console.error(e);
        });
    }
  },
  mounted() {
    this.retrieveParts();
  }
};
```

### Example 2: Navigating After Create

```javascript
savePart() {
  const data = {
    name: this.part.name,
    applicationVehicle: this.part.applicationVehicle,
    netWeight: this.part.netWeight,
    grossWeight: this.part.grossWeight
  };

  PartDataService.create(data)
    .then(response => {
      this.submitted = true;
      // Could redirect to parts list:
      // this.$router.push({ name: 'parts' });
    })
    .catch(e => {
      console.error(e);
    });
}
```

### Example 3: Working with Route Parameters

```javascript
export default {
  name: "part",
  methods: {
    getPart(id) {
      PartDataService.get(id)
        .then(response => {
          this.currentPart = response.data;
        })
        .catch(e => {
          console.error(e);
        });
    }
  },
  mounted() {
    // Get ID from route parameter
    this.getPart(this.$route.params.id);
  }
};
```
